    <script>
        let state = {c:null, targets:[]};
        const allAIs = ['Copilot', 'Perplexity', 'Gemini', 'ChatGPT', 'Claude', 'Genspark'];
        const urlMap = {
            'ChatGPT': 'https://chat.openai.com/', 'Claude': 'https://claude.ai/',
            'Gemini': 'https://gemini.google.com/app', 'Perplexity': 'https://www.perplexity.ai/',
            'Copilot': 'https://copilot.microsoft.com/', 'Genspark': 'https://www.genspark.ai/'
        };

        function clickCat(c, el) {
            state.c = c; state.targets = [];
            document.querySelectorAll('button').forEach(b => { if(b.id !== 'exec') b.classList.remove('active'); });
            el.classList.add('active');
            document.querySelectorAll('.opt-group, .sub-group').forEach(g => g.style.display = 'none');
            const tg = document.getElementById('opt-' + c);
            if(tg) tg.style.display = 'block';
            checkReady();
        }

        function showSub(id, el) {
            el.classList.toggle('active');
            document.getElementById(id).style.display = (document.getElementById(id).style.display === 'block') ? 'none' : 'block';
        }

        // 「全員」ボタン：下の AI 全てを反転させる
        function toggleAllAI(el) {
            el.classList.toggle('active');
            const sub = document.getElementById('B-sub');
            sub.style.display = 'block'; // 強制展開
            document.getElementById('btn-sub-trigger').classList.add('active');

            if (el.classList.contains('active')) {
                state.targets = [...allAIs];
                document.querySelectorAll('.s-btn').forEach(b => b.classList.add('active'));
            } else {
                state.targets = [];
                document.querySelectorAll('.s-btn').forEach(b => b.classList.remove('active'));
            }
            checkReady();
        }

        function toggleOpt(o, el) {
            el.classList.toggle('active');
            if (state.targets.includes(o)) state.targets = state.targets.filter(t => t !== o);
            else state.targets.push(o);
            
            // 全員選ばれたら「全員」ボタンも白くする（連動）
            const allBtn = document.getElementById('btn-all');
            if (state.targets.length === allAIs.length) allBtn.classList.add('active');
            else allBtn.classList.remove('active');
            
            checkReady();
        }

        function checkReady() {
            const btn = document.getElementById('exec');
            if (state.c && state.targets.length > 0) {
                btn.classList.add('ready');
                btn.innerText = "LAUNCH MISSION TO " + state.targets.length + " AIs";
            } else { btn.classList.remove('ready'); btn.innerText = "SELECT TARGETS"; }
        }

        async function fire() {
            if(!state.c || state.targets.length === 0) return;
            state.targets.forEach(t => { window.open(urlMap[t], t); });
            for (const t of state.targets) {
                await fetch('/ask', { method:'POST', headers:{'Content-Type':'application/json'}, 
                    body:JSON.stringify({c: state.c, o: t})
                });
            }
            document.getElementById('log').innerHTML = "> 配送指令完了。各タブを巡回してください。";
        }

        setInterval(async () => {
            const r = await fetch('/get_history');
            const d = await r.json();
            if(!d.length) return;
            let h = '<table><tr><th>Date</th><th>Action</th><th>Text</th></tr>';
            d.reverse().forEach(row => h += '<tr><td>'+row.日付+'</td><td>'+row.名称+'</td><td>'+row.簡単な内容+'</td></tr>');
            document.getElementById('history').innerHTML = h + '</table>';
        }, 5000);
    </script>
